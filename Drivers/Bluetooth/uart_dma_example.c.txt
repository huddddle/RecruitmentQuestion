#include "bluetooth.h"
#include "ti_msp_dl_config.h"

// 示例：在主循环中使用DMA发送
void Example_MainLoop(void)
{
    // 假设的角度和PID参数
    float current_angle = 0.0f;
    float kp = 1.5f, ki = 0.1f, kd = 0.5f;

    // 初始化蓝牙（包含DMA初始化）
    Bluetooth_Init();

    while (1) {
        // ========== 高频控制逻辑 ==========
        // 这里执行你的平衡控制算法
        // 读取传感器、计算PID、控制电机等
        // ...

        // ========== 低频数据发送 ==========
        // 尝试发送数据（非阻塞）
        // 如果上一次发送还没完成，这次会返回false，不影响控制循环
        if (Bluetooth_SendAnglePID_DMA(current_angle, kp, ki, kd)) {
            // 发送成功启动
        } else {
            // 上一次发送还没完成，跳过本次发送
            // 这不会影响控制逻辑的执行
        }

        // 继续执行其他任务...
    }
}

// 示例：在定时器中断中使用DMA发送
void Example_TimerInterrupt(void)
{
    static uint32_t send_counter = 0;
    float current_angle = 0.0f;
    float kp = 1.5f, ki = 0.1f, kd = 0.5f;

    // 每100次控制循环发送一次数据（降低发送频率）
    send_counter++;
    if (send_counter >= 100) {
        send_counter = 0;

        // 尝试发送（非阻塞）
        Bluetooth_SendAnglePID_DMA(current_angle, kp, ki, kd);
    }
}

// 示例：发送自定义字符串
void Example_SendCustomString(void)
{
    const char *msg = "System Ready\r\n";

    // 非阻塞发送
    if (Bluetooth_SendString_DMA(msg)) {
        // 发送成功启动
    } else {
        // 发送失败（上一次未完成）
    }
}

// 示例：等待发送完成
void Example_WaitForComplete(void)
{
    float angle = 45.5f;
    float kp = 1.5f, ki = 0.1f, kd = 0.5f;

    // 启动发送
    Bluetooth_SendAnglePID_DMA(angle, kp, ki, kd);

    // 如果需要确保发送完成再继续（不推荐在高频循环中使用）
    while (!Bluetooth_IsTxComplete()) {
        // 等待发送完成
        // 可以在这里执行其他不依赖串口的任务
    }
}

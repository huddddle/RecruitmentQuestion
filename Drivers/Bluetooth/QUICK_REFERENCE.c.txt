/**
 * ============================================================================
 * UART DMA发送 - 快速参考
 * ============================================================================
 *
 * 用途：实时传输小车角度值和PID参数，最小化CPU占用
 *
 * ============================================================================
 * 初始化（在main函数中）
 * ============================================================================
 */

#include "bluetooth.h"

int main(void) {
    SYSCFG_DL_init();
    Bluetooth_Init();  // 初始化DMA发送

    while(1) {
        // 你的控制循环
    }
}

/**
 * ============================================================================
 * 基本使用 - 发送角度和PID数据
 * ============================================================================
 */

// 方式1：非阻塞发送（推荐）
float angle = 45.5f;
float kp = 1.5f, ki = 0.1f, kd = 0.5f;

if (Bluetooth_SendAnglePID_DMA(angle, kp, ki, kd)) {
    // 发送成功启动
} else {
    // 上一次发送未完成，跳过本次
}

// 方式2：发送自定义字符串
Bluetooth_SendString_DMA("System Ready\r\n");

// 方式3：检查发送状态
if (Bluetooth_IsTxComplete()) {
    // 可以发送下一条数据
}

/**
 * ============================================================================
 * 推荐使用模式 - 在高频控制循环中
 * ============================================================================
 */

void ControlLoop_1kHz(void) {
    static uint32_t counter = 0;

    // 高频控制逻辑（1kHz）
    float angle = ReadSensor();
    ControlMotor(angle);

    // 低频数据发送（10Hz = 每100次发送一次）
    if (++counter >= 100) {
        counter = 0;
        Bluetooth_SendAnglePID_DMA(angle, kp, ki, kd);
    }
}

/**
 * ============================================================================
 * 输出格式
 * ============================================================================
 *
 * Bluetooth_SendAnglePID_DMA(45.5, 1.5, 0.1, 0.5) 输出：
 * Angle=45.50,Kp=1.500,Ki=0.100,Kd=0.500\r\n
 *
 * ============================================================================
 * 性能对比
 * ============================================================================
 *
 * 发送方式      | CPU占用 | 说明
 * -------------|---------|----------------------------------
 * 阻塞发送      | ~100%   | CPU忙等待，严重影响控制
 * 中断发送      | ~30%    | 每字节一次中断
 * DMA+FIFO发送 | <5%     | 只在开始和结束时中断（推荐）
 *
 * ============================================================================
 * 注意事项
 * ============================================================================
 *
 * 1. 单次发送最大64字节
 * 2. 建议发送频率 ≤ 20Hz（取决于波特率9600）
 * 3. 发送函数返回false表示上一次未完成，可以跳过本次
 * 4. 不要在发送完成前修改数据缓冲区
 *
